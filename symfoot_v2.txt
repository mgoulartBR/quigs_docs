Function LoadLeague:Void()
	MyLeagueFile.Open(leaguePath)
	leagueName = MyLeagueFile.ReadString
	totalTeams = CInt(MyLeagueFile.ReadString)
	i = 0
	While i < 32
		If MyLeagueFile.EOF = 0 Then
			gParseLine = MyLeagueFile.ReadString
			gParsePos = 1
			cName(i) = ParseNextField()
			cReputation(i) = CInt(ParseNextField())
			cStadium(i) = CInt(ParseNextField())
			cTicket(i) = CInt(ParseNextField())
			cPaper(i) = CInt(ParseNextField())
			cPen(i) = CInt(ParseNextField())
			cDiv(i) = i / 8
			cMoney(i) = cReputation(i) * 60 + 1000
			cDebt(i) = 0
			cPoints(i) = 0
			cWins(i) = 0
			cDraws(i) = 0
			cLosses(i) = 0
			cGoalsFor(i) = 0
			cGoalsAg(i) = 0
			cMatches(i) = 0
			LoadTeamPlayers
		EndIf
		i = i + 1
	Wend
	rCount = 0
	While rCount < 18
		If MyLeagueFile.EOF = 0 Then
			gParseLine = MyLeagueFile.ReadString
			gParsePos = 1
			rName(rCount) = ParseNextField()
			rReputation(rCount) = CInt(ParseNextField())
			rStadium(rCount) = CInt(ParseNextField())
			rTicket(rCount) = CInt(ParseNextField())
			SkipReservePlayers
			rCount = rCount + 1
		Else
			rCount = 18
		EndIf
	Wend
	MyLeagueFile.Close
EndF
Function LoadTeamPlayers:Void()
	k = i * 18
	j = 0
	While j < 18
		If MyLeagueFile.EOF = 0 Then
			gParseLine = MyLeagueFile.ReadString
			gParsePos = 1
			gPIdx = k
			tmpStr = ParseNextField()
			SetPName
			gPIdx = k
			tmp = CInt(ParseNextField())
			SetPPos
			gPIdx = k
			tmp = CInt(ParseNextField())
			SetPAge
			tmp2 = cReputation(i) / 3 + 10
			tmp = tmp2 + Rnd(35)
			If tmp > 99 Then
				tmp = 99
			EndIf
			If tmp < 1 Then
				tmp = 1
			EndIf
			gPIdx = k
			SetPPts
			gPIdx = k
			tmp = GetPPts() / 10 + 1
			SetPSalary
			gPIdx = k
			tmp = 0
			SetPGols
			gPIdx = k
			tmp = 0
			SetPSusp
			gPIdx = k
			tmp = 0
			SetPStarter
		EndIf
		k = k + 1
		j = j + 1
	Wend
EndF
Function SkipReservePlayers:Void()
	j = 0
	While j < 18
		If MyLeagueFile.EOF = 0 Then
			tmpStr = MyLeagueFile.ReadString
		EndIf
		j = j + 1
	Wend
EndF
Function ParseNextField:String()
	Var result:String
	Var ch:String
	result = ""
	While gParsePos <= Len(gParseLine)
		ch = Mid(gParseLine, gParsePos, 1)
		gParsePos = gParsePos + 1
		If ch = "," Then
			Return result
		Else
			result = result + ch
		EndIf
	Wend
	Return result
EndF

Function InitDivCalendar:Void()
	Var idx:Int
	gDivBase = gDiv * 8
	gCalIdx = gDiv * 56
	r = 0
	While r < 7
		j = 0
		While j < 7
			rot(j) = gDivBase + ((j + r) Mod 7) + 1
			j = j + 1
		Wend
		idx = gCalIdx + r * 4
		If r Mod 2 = 0 Then
			calHome(idx) = gDivBase
			calAway(idx) = rot(6)
		Else
			calHome(idx) = rot(6)
			calAway(idx) = gDivBase
		EndIf
		calHome(idx + 1) = rot(0)
		calAway(idx + 1) = rot(5)
		calHome(idx + 2) = rot(1)
		calAway(idx + 2) = rot(4)
		calHome(idx + 3) = rot(2)
		calAway(idx + 3) = rot(3)
		r = r + 1
	Wend
	i = 0
	While i < 28
		calHome(gCalIdx + 28 + i) = calAway(gCalIdx + i)
		calAway(gCalIdx + 28 + i) = calHome(gCalIdx + i)
		i = i + 1
	Wend
	i = gCalIdx
	While i < gCalIdx + 56
		calGoalH(i) = 0
		calGoalA(i) = 0
		calPlayed(i) = 0
		i = i + 1
	Wend
EndF

Function AutoSelectTeam:Void()
	gBase = t * 18
	i = gBase
	While i < gBase + 18
		gPIdx = i
		tmp = 0
		SetPStarter
		i = i + 1
	Wend
	gPos = 0
	gNeed = 1
	SelectBestN
	gPos = 1
	gNeed = fDef(formation)
	SelectBestN
	gPos = 2
	gNeed = fMid(formation)
	SelectBestN
	gPos = 3
	gNeed = fAtt(formation)
	SelectBestN
EndF
Function SelectBestN:Void()
	Var count:Int
	Var best:Int
	Var bestIdx:Int
	count = 0
	While count < gNeed
		best = -1
		bestIdx = -1
		j = gBase
		While j < gBase + 18
			gPIdx = j
			If GetPPos() = gPos Then
				gPIdx = j
				If GetPStarter() = 0 Then
					gPIdx = j
					If GetPSusp() = 0 Then
						gPIdx = j
						If GetPPts() > best Then
							gPIdx = j
							best = GetPPts()
							bestIdx = j
						EndIf
					EndIf

Function CalcValue:Int()
	Var base:Int
	Var factor:Int
	base = gCalcPts * gCalcPts
	If gCalcAge <= 24 Then
		factor = 12
	EndIf
	If gCalcAge >= 25 Then
		If gCalcAge <= 28 Then
			factor = 10
		EndIf
	EndIf
	If gCalcAge >= 29 Then
		If gCalcAge <= 31 Then
			factor = 7
		EndIf
	EndIf
	If gCalcAge >= 32 Then
		If gCalcAge <= 34 Then
			factor = 4
		EndIf
	EndIf
	If gCalcAge >= 35 Then
		factor = 2
	EndIf
	Return base * factor / 10

Var gHomeT:Int
Var gAwayT:Int
Var gGoalH:Int
Var gGoalA:Int
Function SimulateDivRound:Void()
	Var forceH:Int
	Var forceA:Int
	Var diff:Int
	Var chH:Int
	Var chA:Int
	gCalIdx = gDiv * 56
	gIdx = gCalIdx + currentRound * 4
	i = gIdx
	While i < gIdx + 4
		gHomeT = calHome(i)
		gAwayT = calAway(i)
		gCalcTeam = gHomeT
		forceH = CalcForce() + 5
		gCalcTeam = gAwayT
		forceA = CalcForce()
		diff = forceH - forceA
		chH = 30 + diff
		If chH < 10 Then
			chH = 10
		EndIf
		If chH > 60 Then
			chH = 60
		EndIf
		gGoalH = 0
		j = 0
		While j < 5
			If Rnd(100) < chH Then
				gGoalH = gGoalH + 1
			EndIf
			j = j + 1
		Wend
		chA = 30 - diff
		If chA < 10 Then
			chA = 10
		EndIf
		If chA > 60 Then
			chA = 60
		EndIf
		gGoalA = 0
		j = 0
		While j < 5
			If Rnd(100) < chA Then
				gGoalA = gGoalA + 1
			EndIf
			j = j + 1
		Wend
		calGoalH(i) = gGoalH
		calGoalA(i) = gGoalA
		calPlayed(i) = 1
		UpdateMatchResult
		i = i + 1
	Wend
EndF
Function UpdateMatchResult:Void()
	cMatches(gHomeT) = cMatches(gHomeT) + 1
	cMatches(gAwayT) = cMatches(gAwayT) + 1
	cGoalsFor(gHomeT) = cGoalsFor(gHomeT) + gGoalH
	cGoalsFor(gAwayT) = cGoalsFor(gAwayT) + gGoalA
	cGoalsAg(gHomeT) = cGoalsAg(gHomeT) + gGoalA
	cGoalsAg(gAwayT) = cGoalsAg(gAwayT) + gGoalH
	If gGoalH > gGoalA Then
		cWins(gHomeT) = cWins(gHomeT) + 1
		cLosses(gAwayT) = cLosses(gAwayT) + 1
		cPoints(gHomeT) = cPoints(gHomeT) + 3
		If gHomeT = playerTeam Then
			consWins = consWins + 1
			consLosses = 0
		EndIf
		If gAwayT = playerTeam Then
			consLosses = consLosses + 1
			consWins = 0
		EndIf
	EndIf
	If gGoalH = gGoalA Then
		cDraws(gHomeT) = cDraws(gHomeT) + 1
		cDraws(gAwayT) = cDraws(gAwayT) + 1
		cPoints(gHomeT) = cPoints(gHomeT) + 1
		cPoints(gAwayT) = cPoints(gAwayT) + 1
		If gHomeT = playerTeam Then
			consWins = 0
			consLosses = 0
		EndIf
		If gAwayT = playerTeam Then
			consWins = 0
			consLosses = 0
		EndIf
	EndIf
	If gGoalH < gGoalA Then
		cLosses(gHomeT) = cLosses(gHomeT) + 1
		cWins(gAwayT) = cWins(gAwayT) + 1
		cPoints(gAwayT) = cPoints(gAwayT) + 3
		If gHomeT = playerTeam Then
			consLosses = consLosses + 1
			consWins = 0
		EndIf
		If gAwayT = playerTeam Then
			consWins = consWins + 1
			consLosses = 0
		EndIf
	EndIf
EndF
Function GenMatchEvents:Void()
	Var gh:Int
	Var ga:Int
	Var gameRow:Int
	Var scorer:Int
	matchEvCnt = 0
	matchEvShown = 0
	gDiv = 0
	While gDiv < 4
		gCalIdx = gDiv * 56
		gIdx = gCalIdx + currentRound * 4
		gameRow = 0
		While gameRow < 4
			i = gIdx + gameRow
			gh = calGoalH(i)
			ga = calGoalA(i)
			j = 0
			While j < gh
				If matchEvCnt < 80 Then
					matchEvMin(matchEvCnt) = 1 + Rnd(89)
					gCalcTeam = calHome(i)
					scorer = PickScorer()
					gPIdx = scorer
					gPIdx = scorer
					tmp = GetPGols() + 1
					SetPGols
					gPIdx = scorer
					matchEvTxt(matchEvCnt) = "D" + Str(gDiv) + "G" + Str(gameRow) + "H " + Str(matchEvMin(matchEvCnt)) + "' " + GetPName()
					matchEvCnt = matchEvCnt + 1
				EndIf
				j = j + 1
			Wend
			j = 0
			While j < ga
				If matchEvCnt < 80 Then
					matchEvMin(matchEvCnt) = 1 + Rnd(89)
					gCalcTeam = calAway(i)
					scorer = PickScorer()
					gPIdx = scorer
					gPIdx = scorer
					tmp = GetPGols() + 1
					SetPGols
					gPIdx = scorer
					matchEvTxt(matchEvCnt) = "D" + Str(gDiv) + "G" + Str(gameRow) + "A " + Str(matchEvMin(matchEvCnt)) + "' " + GetPName()
					matchEvCnt = matchEvCnt + 1
				EndIf
				j = j + 1
			Wend
			gameRow = gameRow + 1
		Wend
		gDiv = gDiv + 1
	Wend
	SortMatchEvents
EndF
Function PickScorer:Int()
	Var cnt:Int
	Var picks(11):Int
	Var wantPos:Int
	tmp = Rnd(100)
	If tmp < 10 Then
		wantPos = 1
	EndIf
	If tmp >= 10 Then
		If tmp < 50 Then
			wantPos = 2
		EndIf
	EndIf
	If tmp >= 50 Then
		wantPos = 3
	EndIf
	cnt = 0
	k = gCalcTeam * 18
	While k < gCalcTeam * 18 + 18
		gPIdx = k
		If GetPStarter() = 1 Then
			gPIdx = k
			If GetPPos() = wantPos Then
				picks(cnt) = k
				cnt = cnt + 1
			EndIf
		EndIf
		k = k + 1
	Wend
	If cnt = 0 Then
		k = gCalcTeam * 18

Function SortMatchEvents:Void()
	Var swapped:Int
	Var tempMin:Int
	Var tempTxt:String
	swapped = 1
	While swapped = 1
		swapped = 0
		i = 0
		While i < matchEvCnt - 1
			If matchEvMin(i) > matchEvMin(i + 1) Then
				tempMin = matchEvMin(i)
				tempTxt = matchEvTxt(i)
				matchEvMin(i) = matchEvMin(i + 1)
				matchEvTxt(i) = matchEvTxt(i + 1)
				matchEvMin(i + 1) = tempMin
				matchEvTxt(i + 1) = tempTxt

			EndIf
			i = i + 1
		Wend

Function CalcAttendance:Int()
	Var att:Int
	Var divFactor:Int
	tmp2 = gCalcTeam / 8
	If tmp2 = 0 Then
		divFactor = 100
	EndIf
	If tmp2 = 1 Then
		divFactor = 75
	EndIf
	If tmp2 = 2 Then
		divFactor = 55
	EndIf
	If tmp2 = 3 Then
		divFactor = 40
	EndIf
	att = cStadium(gCalcTeam) * 6
	att = att * divFactor / 100
	att = att * (100 - cTicket(gCalcTeam)) / 100
	att = att * cReputation(gCalcTeam) / 100
	If cMatches(gCalcTeam) > 0 Then
		att = att * (50 + cWins(gCalcTeam) * 50 / cMatches(gCalcTeam)) / 100
	EndIf
	att = att + att * (Rnd(21) - 10) / 100
	If att > cStadium(gCalcTeam) * 10 Then
		att = cStadium(gCalcTeam) * 10
	EndIf
	If att < 5 Then
		att = 5
	EndIf
	Return att
EndF

Function ProcessEndRound:Void()
	t = 0
	While t < 32
		gCalcTeam = t
		cMoney(t) = cMoney(t) - CalcPayroll()
		cMoney(t) = cMoney(t) - cDebt(t) * 5 / 100
		cMoney(t) = cMoney(t) + cReputation(t) * 5
		t = t + 1
	Wend
	gDiv = 0
	While gDiv < 4
		gCalIdx = gDiv * 56
		gIdx = gCalIdx + (currentRound - 1) * 4
		i = gIdx
		While i < gIdx + 4
			gCalcTeam = calHome(i)
			tmp = CalcAttendance()
			tmp = tmp * cTicket(calHome(i)) / 10
			cMoney(calHome(i)) = cMoney(calHome(i)) + tmp
			i = i + 1
		Wend
		gDiv = gDiv + 1
	Wend
	i = 0
	While i < 576
		gPIdx = i
		If GetPSusp() > 0 Then
			gPIdx = i
			tmp = GetPSusp() - 1
			SetPSusp
		EndIf
		i = i + 1
	Wend
	UpdatePlayerForm
EndF

Function UpdateTeamForm:Void()
	j = gCalcTeam * 18
	k = j + 18
	While j < k
		gPIdx = j
		If GetPStarter() = 1 Then
			gPIdx = j
			tmp = GetPPts()
			gPIdx = j
			If GetPAge() < 25 Then
				If Rnd(100) < 30 Then
					tmp = tmp + 1
				EndIf
			EndIf
			gPIdx = j
			If GetPAge() > 32 Then
				If Rnd(100) < 20 Then
					tmp = tmp - 1
				EndIf
			EndIf
			If tmp2 > 0 Then
				If Rnd(100) < 15 Then
					tmp = tmp + 1
				EndIf
			EndIf
			If tmp2 < 0 Then
				If Rnd(100) < 10 Then
					tmp = tmp - 1
				EndIf
			EndIf
			If tmp > 99 Then
				tmp = 99
			EndIf
			If tmp < 1 Then
				tmp = 1
			EndIf
			gPIdx = j
			SetPPts
		EndIf
		j = j + 1
	Wend
EndF

		consWins = 0
		consLosses = 0
		bx1 = MsgBox("Welcome to " + cName(playerTeam),"Div " + Str(playerDiv + 1),"",MsgBox.Title.Confirmation,MsgBox.Buttons.Ok,MsgBox.Icon.Default)
		RefreshSquad
		RefreshInfo
	EndIf
EndF
Function OfferNewTeams:Void()
	Var c1:Int
	If playerDiv < 3 Then
		c1 = (playerDiv + 1) * 8 + Rnd(8)
	Else
		c1 = playerDiv * 8 + Rnd(8)
	EndIf
	If c1 = playerTeam Then
		c1 = c1 + 1
		If c1 >= 32 Then
			c1 = 24
		EndIf
	EndIf
	bx1 = MsgBox("Join " + cName(c1) + "?","Div " + Str(c1 / 8 + 1),"",MsgBox.Title.Confirmation,MsgBox.Buttons.YesNo,MsgBox.Icon.Default)
	If bx1 = 3 Then
		playerTeam = c1
		playerDiv = c1 / 8
	Else
		playerTeam = c1
		playerDiv = c1 / 8
	EndIf
	consWins = 0
	consLosses = 0
	bx1 = MsgBox("New team: " + cName(playerTeam),"Div " + Str(playerDiv + 1),"",MsgBox.Title.Confirmation,MsgBox.Buttons.Ok,MsgBox.Icon.Default)
	RefreshSquad
	RefreshInfo
EndF

Function EndSeason:Void()
	Var oldDiv:Int
	Var wasChamp:Int
	oldDiv = playerDiv
	wasChamp = 0
	FindPlayerPosition
	If tmp = 1 Then
		wasChamp = 1
	EndIf
	bx1 = MsgBox("End of Season!","Processing promotions...","",MsgBox.Title.Confirmation,MsgBox.Buttons.Ok,MsgBox.Icon.Default)
	gDiv = 0
	While gDiv < 3
		DoPromotion
		gDiv = gDiv + 1
	Wend
	DoReserveSwap
	playerDiv = playerTeam / 8
	If playerDiv > oldDiv Then
		bx1 = MsgBox("RELEGATED!","You are fired!","",MsgBox.Title.Warning,MsgBox.Buttons.Ok,MsgBox.Icon.Default)
		OfferNewTeams
	EndIf
	If wasChamp = 1 Then
		If oldDiv > 0 Then
			ManagerInvited
		EndIf
	EndIf
	consWins = 0
	consLosses = 0
	ResetSeasonStats
	InitAllCalendars
	AutoSelectAll
	currentRound = 0
	season = season + 1
	AgeAllPlayers
	bx1 = MsgBox("Season " + Str(season),cName(playerTeam) + " - Div " + Str(playerDiv + 1),"",MsgBox.Title.Confirmation,MsgBox.Buttons.Ok,MsgBox.Icon.Default)
	RefreshSquad
	RefreshInfo
EndF
Function DoPromotion:Void()
	FindDivBottom
	relDn1 = sortIdx(6)
	relDn2 = sortIdx(7)
	gDiv = gDiv + 1
	FindDivTop
	proUp1 = sortIdx(0)
	proUp2 = sortIdx(1)
	gDiv = gDiv - 1
	SwapTeamDiv
EndF
Function FindDivBottom:Void()
	gDivBase = gDiv * 8
	i = 0
	While i < 8
		sortIdx(i) = gDivBase + i
		i = i + 1
	Wend
	Var swapped:Int
	swapped = 1
	While swapped = 1
		swapped = 0
		i = 0
		While i < 7
			If cPoints(sortIdx(i)) > cPoints(sortIdx(i + 1)) Then
				tmp = sortIdx(i)
				sortIdx(i) = sortIdx(i + 1)
				sortIdx(i + 1) = tmp
				swapped = 1
			EndIf
			i = i + 1
		Wend
	Wend
EndF
Function FindDivTop:Void()
	gDivBase = gDiv * 8
	i = 0
	While i < 8
		sortIdx(i) = gDivBase + i
		i = i + 1
	Wend
	Var swapped:Int
	swapped = 1
	While swapped = 1
		swapped = 0
		i = 0
		While i < 7
			If cPoints(sortIdx(i)) < cPoints(sortIdx(i + 1)) Then
				tmp = sortIdx(i)
				sortIdx(i) = sortIdx(i + 1)
				sortIdx(i + 1) = tmp
				swapped = 1
			EndIf
			i = i + 1
		Wend
	Wend
EndF

Function SwapOnePair:Void()

	cDebt(relDn1) = cDebt(proUp1)

	si = cStadium(relDn1)

	cPaper(proUp1) = si

Function DoReserveSwap:Void()
	gDiv = 3
	FindDivBottom
	relDn1 = sortIdx(6)
	relDn2 = sortIdx(7)
	If rCount >= 2 Then
		SwapWithReserve
	EndIf
EndF
Function SwapWithReserve:Void()
	Var r1:Int
	Var r2:Int
	Var sn:String
	Var si:Int
	r1 = Rnd(rCount)
	r2 = Rnd(rCount)
	While r2 = r1
		r2 = Rnd(rCount)
	Wend
	If relDn1 = playerTeam Then
	Else
		sn = cName(relDn1)
		si = cReputation(relDn1)
		cName(relDn1) = rName(r1)
		cReputation(relDn1) = rReputation(r1)
		cStadium(relDn1) = rStadium(r1)
		cTicket(relDn1) = rTicket(r1)
		cMoney(relDn1) = rReputation(r1) * 60 + 1000
		cDebt(relDn1) = 0
		cPaper(relDn1) = 13
		cPen(relDn1) = 1
		rName(r1) = sn
		rReputation(r1) = si
		t = relDn1
		GenTeamPlayers
	EndIf
	If relDn2 = playerTeam Then
	Else
		sn = cName(relDn2)
		si = cReputation(relDn2)
		cName(relDn2) = rName(r2)
		cReputation(relDn2) = rReputation(r2)
		cStadium(relDn2) = rStadium(r2)
		cTicket(relDn2) = rTicket(r2)
		cMoney(relDn2) = rReputation(r2) * 60 + 1000
		cDebt(relDn2) = 0
		cPaper(relDn2) = 13
		cPen(relDn2) = 1
		rName(r2) = sn
		rReputation(r2) = si
		t = relDn2
		GenTeamPlayers
	EndIf
EndF
Function ResetSeasonStats:Void()
	i = 0
	While i < 32
		cPoints(i) = 0
		cWins(i) = 0
		cDraws(i) = 0
		cLosses(i) = 0
		cGoalsFor(i) = 0
		cGoalsAg(i) = 0
		cMatches(i) = 0
		i = i + 1
	Wend
	i = 0
	While i < 576
		gPIdx = i
		tmp = 0
		SetPGols
		gPIdx = i
		tmp = 0
		SetPSusp
		i = i + 1
	Wend
EndF
Function AgeAllPlayers:Void()
	i = 0
	While i < 576
		gPIdx = i
		gPIdx = i
		tmp = GetPAge() + 1
		SetPAge
		gPIdx = i
		If GetPAge() >= 42 Then
			t = i / 18
			k = i
			GenOnePlayer
			gPIdx = k
			tmp = 15 + Rnd(25)
			SetPPts
			gPIdx = k
			tmp = 17 + Rnd(4)
			SetPAge
			gPIdx = k
			gPIdx = k
			tmp = GetPPts() / 10 + 1
			SetPSalary
		EndIf
		i = i + 1
	Wend
EndF
