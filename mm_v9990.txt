// Application First Start is Here, set and load all defaults
Function Initialize:Void()
	Var f:Int
	KeyMul.Set Key.Code.Space,Key.Code.CursorLeft,Key.Code.CursorRight,Key.Code.ESC,0,0
	
	// Initialize THIS is  G9K App for SymbOS and use the default MapEngine for the G9k
	MapEngine 1
	Set.Sprites 0,112
	Disable.G9KInterrupt
	// Set the border collor to 0
	// Border Color can be 0-63
	BackColour 0
	
	// There are 16 color sets (or called ink) 0.. 16. 32 .. etc.
	// During load you give the offset of you color set
	// A set can be set to the hardware pallets 0-3
	// Make the screen Black
	// Set the Ink RGB all to 0,0,0
	For f=240 To 255
		Ink f,0,0,0
	Next
	
	// G9K can have 4 pallets of 16 colors (0-3)
	// Set the 4 palletes to black so during load it stays black
	Ink.Palette 0,240
	Ink.Palette 1,240
	Ink.Palette 2,240
	Ink.Palette 3,240
	;
	// By default all sprites are enabled and visiable.
	// G9k can have 124 sprites (16x16) in total
	// We set all sprites to pallette 0 as this is black now
	// We are moving all the sprites to location 0,190 to avoid flickering
	// We set the display for all sprites to 1
	// 0=Visable in front of ForeLayer
	// 1=Not Visable
	// 2=Visable Behind ForeLayer
	// 3=Visable in front of BackLayer
	
	For f=0 To 124
		Palette.Sprite f,0
		Draw.Sprite f,0,0
		Display.Sprite f,1
	Next
	Update.Display
	
	// Loading bitmap pixel sheet data in
	// First we will load the sprite sheet starting at memory block 0 using palleteOffset 48
	pattCount=Load.Pattern.A(App.Path+"mmm001.qpt",0,48) // Sprite Sheet Bitmap
	// We used pallet 2 for the sprite bitmap sheet
	Ink.Palette 2,48
	
	// Sprite 0 is the mouse pointer sprite. This is common to do as this sprite is on top of the other sprites
	// For every sprite we set the pallet, Position and Pattern (Bitmap) data.
	// Sprite 0  = Mouse
	// Spite 99 = Exit button to Exit
	
	// We use the waiting glass for loading
	Draw.Sprite 0,128,106
	Pattern.Sprite 0,glass
	Palette.Sprite 0,2
	Display.Sprite 0,0
	
	Draw.Sprite 99,245,1
	Pattern.Sprite 99,0
	Palette.Sprite 99,2
	
	//Waiting picture in the center
	Draw.Sprite 116,120,130
	Draw.Sprite 117,136,130
	Pattern.Sprite 116,16
	Pattern.Sprite 117,17
	Palette.Sprite 116,2
	Palette.Sprite 117,2
	Display.Sprite 116,0
	Display.Sprite 117,0
	
	Update.Display
	
	WaitingGlassLoadingAnimation
	
	//sound daemon is active load sounds
	
	If Sound.Hardware =0 Then
		//No sound hardware to play
		soundhardware=0
	ElseIf Sound.Hardware=1 Then
		//PSG is prefered
		soundhardware=1
		SongHandle=Sound.Music.LoadPSG(App.Path +"mmmusic.spm")
		WaitingGlassLoadingAnimation
		EffectHandle=Sound.FX.LoadPSG(App.Path +"mmmfx.spx")
	ElseIf Sound.Hardware=3 Then
		//OPL is preferd
		soundhardware=2
		SongHandle=Sound.Music.LoadOPL4(App.Path +"mmmusic.swm")
		WaitingGlassLoadingAnimation
		EffectHandle=Sound.FX.LoadOPL4(App.Path +"mmmfx.swx")
	EndIf
	
	//Now we need to load the sheet bitmap pattern data in for the Fore Layer and Back Layer
	pattCount=Load.Pattern.B(App.Path+"mmm002.qpt",1024,128)				;was 0 Palette at 128
	WaitingGlassLoadingAnimation
	pattCount=Load.Pattern.A(App.Path+"mmm003.qpt",4096,144)				;was 1024Palette at 144
	WaitingGlassLoadingAnimation
	
	// Load the MAP structure in which is desgined in Quigs.
	// MAP contains the structure and is build out of patterns from the sheets
	// There is also a META file which contains the properties for each pattern.
	
	WaitingGlassLoadingAnimation
	If Meta.Load(App.Path+"map1.qmt")=0 Then
		ShutdownApp
	EndIf
	WaitingGlassLoadingAnimation
	If Load.MapB(App.Path+"map1.qmb")=0 Then
		ShutdownApp
	EndIf
	WaitingGlassLoadingAnimation
	If Load.MapA(App.Path+"map1.qma")=0 Then
		ShutdownApp
	EndIf
	
	// Create a new pallet white in RGB color
	For f=224 To 239
		Ink f,31,31,31
	Next
	
	
	
	// The Drawmap function will draw the visable screen to the location on the MAP you set
	// We set both layers to top left but because we use still a black pallet nothing is seen
	DrawBackMap (0,0)
	DrawForeMap (0,0)
	
	// Do not show the mouse anymore but set the normal mouse icon and show it again after the small intro
	Display.Sprite 0,1
	Pattern.Sprite 0,2
	
	// Some General Functions to initialize some sprites
	LoadMenuSprites
	LoadMenuSprites_PressFire
	LoadMenuSprites_Select
	LoadMenuSprites_Cursor
	LoadMenuSprites_GetReady
	
	// The load of the program is complet
	// Now the MAIN is called and called everytime when it gets CPU time from SymbOS
	
	Display.Sprite 116,1
	Display.Sprite 117,1
	
	//Set some Meta Tile Data for the characters
	character(0) = 896  	//Bruno
	character(1) = 906  	//Spider
	character(2) = 916  	//Joline
	character(3) = 926  	//Edina
	character(4) = 936  	//Cherry
	character(5) = 946  	//Emile
	character(6) = 768  	//Ray
	character(7) = 778  	//Sara
	Enable.G9KInterrupt
	
	
	
	//Set the Ink  for the Fore Layer  and Back Layer
	Ink.Palette 0,240					//This is the Ink 0 we're going To use For Background
	Ink.Palette 1,240					//This is the Ink 1 we're using For foreground
	
	// Now set the pallete to both Layers (We set it to 240 because we want a Fade)
	Palette.LayerB 0					//As above set back as 0, fore as 1
	Palette.LayerA 1
	Update.Display
	
	
	fade=0
	menustatus=1
	menuactive=1
EndF

// MAIN LOOP
// -------------------------------------------------------------------------------------------------------------------------------------
Function Main:Void()
	// Main loop called everytime when it gets CPU time from the SymbOS kernal.
	// Every loop do a sprite refresh
	
	// The Menu Structure
	// If the menu is on and above 0 then it will check for Menu actions
	
	If menuactive=1 Then
		CheckKeys
		menu
	Else
		GameLoop
	EndIf
EndF
// -------------------------------------------------------------------------------------------------------------------------------------

Function CheckKeys:Void()
	//KeyMul.Set Key.Code.Space,Key.Code.CursorLeft,Key.Code.CursorRight,Key.Code.ESC,0,0
	KeyMul.Read
	If KeyMul.Get(KeyMul.Scan1) Then
		SpaceKey=Min(SpaceKey+1,2)
	Else
		SpaceKey=0
	EndIf
	If KeyMul.Get(KeyMul.Scan4) Then
		ESCKey=Min(ESCKey+1,2)
	Else
		ESCKey=0
	EndIf
	If KeyMul.Get(KeyMul.Scan2)  Then
		If arrowLeftKey=0 Then
			arrowLeftKey=1
		EndIf
	Else
		arrowLeftKey=0
	EndIf
	If KeyMul.Get(KeyMul.Scan3)  Then
		If arrowRightKey=0 Then
			arrowRightKey=1
		EndIf
	Else
		arrowRightKey=0
	EndIf
EndF

Function PlayMusic:Void(song)
	If soundhardware=0 Then
		//no sound
	ElseIf soundhardware=1 Then
		Sound.Music.Play SongHandle,0
	ElseIf soundhardware=2 Then
		Sound.Music.Play SongHandle,0
	EndIf
EndF


Function PlayEffect:Void(effect,Vol)
	// SoundFX OPL4 File
	//0=Select (move  in menu)
	//1=Selected item or Powerup
	//2= Signal for start (play it 3 times and than 1x longer
	//3=Horn/Claxon1
	//4=Horn/Claxon2
	//5=Fail of the board
	//6=Jump
	//7=Menu not an option or in game not an option sound
	//8=User player overtakes the computer and goes one place up
	//9=Computer player takesover player  and foes one place down
	//10=Level complete (going over the finisch line sound) (car goes over the line and start jumping faster on the same position.)
	//11=stearing sound / slipping sound
	//12=Bump into hard object
	//13=Bump into softer object
	//14 You are slipping 360 over water of oli)
	//15=driving over a small object
	//16=driving into other car  sound1 (harder)
	//17=the crash sound after you are on the ground
	//18=dirving into other car sound2 (softer)
	//19= Motor sound1 (more soft)
	//20=Motor sound2 (more hard)
	//21=Motor sound3 (more middle hard)
	//22=Powerup
	
	If soundhardware=0 Then
		//no sound
	ElseIf soundhardware=1 Then
		Sound.FX.Play EffectHandle,effect,Vol
	ElseIf soundhardware=2 Then
		Sound.FX.PlayOPL4 EffectHandle,effect,Vol,0
	EndIf
EndF


// ALL MENU FUNCTONS CALLED FROM THE MAIN
Function menu:Void
	Var f:Int
	If menustatus > 0 Then
		// The menu is in status 6 that means we can show a mouse if the user moves it.
		If  menustatus >6 Then
			//mouseloop
			// The animations are running and background is moving
			If menustatus=7 Then
				// Status 7 is the first interactive Menu Screen
				menu7
			ElseIf menustatus=8 Then
				// User pressed Fire and the Car Selector is loaded
				menu8
			ElseIf menustatus=9 Then
				// Second stage,  after Car select, user select player to play
				menu9
			ElseIf menustatus=10 Then
				// Car and Players are Selected and showed before loading the real game MAP
				menu10
			ElseIf menustatus=11 Then
				// Results of playing
				menu11
			ElseIf menustatus=99 Then
				// Results of playing
				menu99
			EndIf
		Else
			// If there is a fade request to fade the screen in or out the routine is called to do
			If fade>0 Then
				fadeLoop
			EndIf
			// The first status of the boot of the game.  It will activate the FADE in
			If menustatus=1 Then
				If fade=0 Then
					// The logo is already on the correct position in Display set in the initialize
					menu1
				EndIf
			ElseIf menustatus=2 Then
				If fade=0 Then
					// Status 2 and 5 are used to wait for a while after the logo is diplayed
					menu2
				EndIf
			ElseIf menustatus=3 Then
				If fade=0 Then
					// Fade everthing back to black again
					menu3
				EndIf
			ElseIf menustatus=4 Then
				If  fade=0 Then
					// Show the Second Quigs Logo on the screen
					menu4
				EndIf
			ElseIf menustatus=5 Then
				If fade=0 Then
					// Status 2 and 5 are used to wait for a while after the logo is diplayed
					menu2
				EndIf
			ElseIf menustatus=6 Then
				If fade=0 Then
					// Move to the first MMM intro screen and set everyting correct
					PlayMusic(0)
					menu6
				EndIf
			EndIf
		EndIf
	EndIf
	
	//	When ESC is pressed by the user it will go to to the MENU
	//	Second press will Exit the Application and go back to the SymbOS desktop
	If ESCKey=1  And menustatus < 90 Then
		PlayEffect(7,255)
		//clear all sprites and set them
		For f=0 To 124
			Display.Sprite f,1
		Next
		Update.Display
		If menustatus=7 Then
			menustatus=99
			EnableExit
		Else
			menustatus=7
		EndIf
		menu6
		HidePressFire
		hideGetReady
		hideSelect
		ShowMMLogo
		Update.Display
		Ink.Palette 0,128					//This is the Ink 0
		Ink.Palette 1,144					//This is the Ink 1
		
	EndIf
EndF

// Menu 99 is the last for exit back to the symbos desktop (it enabled the mouse)
Function menu99:Void()
	mouseloop
	menu7
	If ESCKey=1 Then
		PlayEffect(7,255)
		ShutdownApp
	EndIf
	If SpaceKey=1  Then
		PlayEffect(1,255)
		menustatus=7
		HideCursor
		LoadMenuSprites
	EndIf
EndF

// Routine to shutdown App
Function ShutdownApp:Void()
	If soundhardware > 0 Then
		If  soundhardware > 0 Then
			Sound.Music.Release SongHandle
			Sound.FX.Release EffectHandle
		EndIf
	EndIf
	App.Exit
EndF

Function menu1:Void
	// Activate FADE to show the First Quigs Logo (1= Fade IN, 2=Fade OUT)
	fade=1
	fadeset=240
	menustatus	=2
EndF

Function menu2:Void
	// Wait for 50 counter loops and then set the Next menu status.
	timercount=timercount+1
	If timercount => 50 Then
		If menustatus=2 Then
			menustatus=3
		EndIf
		If menustatus=5 Then
			fade=2
			fadeset=240
			menustatus=6
		EndIf
		timercount=0
	EndIf
EndF

Function menu3:Void
	// Fade out the Quigs logo
	fade=2
	fadeset=240
	menustatus	=4
EndF

Function menu7:Void
	// Activate the Backscroll and Helmet Animation
	MenuBackScroll
	LoadMenuSprites_AnimateHemlet
	// Wait for the user to press space to go to the next phase
	If menustatus=99 Then
	Else
		If SpaceKey=1  Then
			PlayEffect(1,255)
			Ink.Palette 0,240
			Ink.Palette 1,240
			Update.Display
			// Remove text press fire
			HidePressFire
			// Hide the Helms
			Display.Sprite 97,1
			Display.Sprite 98,1
			// Show  cursor sprites
			ShowCursor

			menustatus=8
			cartype=0
			player(0)=0
			Update.Display
			Ink.Palette 0,128
			Ink.Palette 1,144
		EndIf
	EndIf

	BlinkTimer	(1)
EndF

Function menu8:Void
	// Car Select
	MenuBackScroll
	BlinkTimer	(2)
	// Scroll the cars step by step, scroll to the target
	If fX=xTarget Then
		If arrowRightKey=1 Then
			cartype=cartype+1
			If cartype=>5 Then
				cartype=5
			EndIf
			xTarget=xTarget+102
			If xTarget>1305  Then
				xTarget=1305
			EndIf
			scrolldirection=1
			PlayEffect(0,255)
		EndIf
		If arrowLeftKey=1  Then
			cartype=cartype-1
			If cartype=<0 Then
				cartype=0
			EndIf
			
			xTarget=xTarget-102
			If xTarget<795 Then
				xTarget=795
			EndIf
			scrolldirection=2
			PlayEffect(0,255)
		EndIf
		If SpaceKey=1 Then
			Ink.Palette 0,240
			Ink.Palette 1,240
			Update.Display
			PlayEffect(1,255)
			menustatus=9
			DrawForeMap(784,180)
			Update.Display
			Ink.Palette 0,128
			Ink.Palette 1,144
			fX=784
			xTarget=784
		EndIf
	Else
		If scrolldirection=1 Then
			fX=Min(fX+6,xTarget)
		ElseIf scrolldirection=2 Then
			fX=Max(fX-6,xTarget)
		EndIf
		Scroll.LayerA fX,Scroll.LayerA.Y()
		//Car is slected go to the next stage (Choose character/player)
	EndIf
	//Arrow animation
	Draw.Sprite 96,120,190-Shr4(IntSin(Shl2(count) And 127))
EndF

Function menu9:Void
	// Player Select
	MenuBackScroll
	BlinkTimer	(2)
	// Scroll the Player characters  step by step, scroll to the target
	If fX=xTarget Then
		If arrowRightKey =1 Then
			player(0)=player(0)+1
			If player(0)=>7 Then
				player(0)=7
			EndIf
			xTarget=xTarget+64
			If xTarget>1232  Then
				xTarget=1232
			EndIf
			scrolldirection=1
			PlayEffect(0,255)
		EndIf
		If arrowLeftKey =1 Then
			player(0)=player(0)-1
			If player(0)=<0 Then
				player(0)=0
			EndIf
			
			xTarget=xTarget-64
			If xTarget<784 Then
				xTarget=784
			EndIf
			scrolldirection=2
			PlayEffect(0,255)
		EndIf
		//Go to the Next Menu (SHOW GET READY SCREEN, before load game state)
		If SpaceKey=1 Then
			PlayEffect(1,255)
			Ink.Palette 0,240
			Ink.Palette 1,240
			Update.Display
			DrawForeMap(1792,0)
			hideMMLogo
			HideCursor
			hideSelect
			menustatus=10
			Var tempnumber:Int
			Randomize Time.Second.Get
			numbers ="01234567"
			numbers =Replace (numbers,Str(player(0)),"")
			While Len(numbers)=3
				tempnumber=Rnd(8)
				numbers =Replace (numbers,Str(tempnumber),"")
			Wend
			
			player(1) = CInt(Mid(numbers,1,1))
			player(2) = CInt(Mid(numbers,2,1))
			player(3) = CInt(Mid(numbers,3,1))
			anitimer=20
			// Remove Arrow sprite
			Display.Sprite	 96,1
			Update.Display
			Ink.Palette 0,128
			Ink.Palette 1,144
		EndIf
	Else
		If scrolldirection=1 Then
			fX=Min(fX+6,xTarget)
		EndIf
		
		If scrolldirection=2 Then
			fX=Max(fX-6,xTarget)
		EndIf
		Scroll.LayerA fX,Scroll.LayerA.Y()
	EndIf
	//Arrow Animation
	Draw.Sprite 96,120,190-Shr4(IntSin(Shl2(count) And 127))
EndF

Function fadeLoop:Void
	If fade=1 Then				;Fading up
		fadevalue=Min(fadevalue+4,255)
		PalFade(128,fadeset,0,fadevalue)		//(128)Back palette,(240)Black palette,(0)destination palette
		PalFade(144,fadeset,16,fadevalue)		//(128)Fore palette,(240) Black palette,(16)destination palette
		Ink.Palette 0,0
		Ink.Palette 1,16
		If fadevalue=255 Then
			fade=0					//Okay we've reached top palette set fade too off
		EndIf
		Else					;Fading Down
		fadevalue=Max(fadevalue-4,0)
		PalFade(128,fadeset,0,fadevalue)		//(128)Back palette,(240)Black palette,(0)destination palette
		PalFade(144,fadeset,16,fadevalue)		//(128)Fore palette,(240) Black palette,(16)destination palette
		Ink.Palette 0,0
		Ink.Palette 1,16
		If fadevalue=0 Then
			fade=0					//Okay we've reached top palette set fade too off
		EndIf
	EndIf
EndF

Function mouseloop:Void
	// Mouse Routine, Only called when exit the game
	Mouse.XYKey
	MouseOk=0
	If MouseX<>Mouse.Result.X Then
		MouseOk=1
	ElseIf Mouse.Result.Key>0 Then
		MouseOk=1
	ElseIf  MouseY<>Mouse.Result.Y Then
		MouseOk=1
	EndIf
	If MouseOk=1 Then
		// mouse should not go out of the window
		MouseX = Max(3,Min(Mouse.Result.X,250))
		MouseY = Max(3,Min(Mouse.Result.Y,203))
		If Mouse.Result.Key=1 Then
			ShutdownApp
		EndIf
		If Mouse.Result.Key=2 Then
			LoadMenuSprites
		EndIf
		Draw.Sprite 0,MouseX,MouseY
		MouseCount=0
		Display.Sprite 0,0
		Display.Sprite 99,0
	Else
		MouseCount=MouseCount+1
		If MouseCount=50 Then
			Display.Sprite 0,1
			Display.Sprite 99,1
		EndIf
	EndIf
	
EndF

Function EverythingBlack:Void()
	For f=0 To 15
		Ink f,0,0,0
	Next
	Ink.Palette 0,0
	Palette.LayerB 0
	Palette.LayerA 0
	;Clear those Spites
	For f=0 To 120
		Display.Sprite f,1
		Draw.Sprite f,256,228
	Next
EndF

Function InitGame:Void()
	// Set Everything to Black
	
	EverythingBlack
	Update.Display
	Disable.Sprites
	Disable.G9KInterrupt
	
	;Load_Level(1,0)					//Map m5, vehicle 4x4
	;PosAngle_Vehicles(392,776,0)		// Location of cars & angle
	
	Load_Level(2,1)					//Map m4, vehicle 4x4
	PosAngle_Vehicles(176,1080,0)		// Location of cars & angle
	
	
	e=0
	a=0
	c=0
	b=0
	d=0
	a=Range(MapXMin,MapXMax,Player_X-128)
	b=Range(MapYMin,MapYMax,Player_Y-100)
	
	Scroll.LayerA a-1,b-1
	Draw.LayerA
	a=Shr2(a):b=Shr2(b)
	Scroll.LayerB a-1,b-1
	Draw.LayerB
	
	// Helmet positions
	For f=0 To 3
		Pos(f)=0
	Next
	PosNx(0)=57:PosNx(1)=12:PosNx(2)=27:PosNx(3)=42
	PosTrigger=0
	DrawOddSprites								// Draw Helmet / Lap / Timer / Result Sprites on screen
	
	// Level Palette
	Ink.Palette 0,32://Palette.LayerA 0					// Palette for Layer A = 0
	Ink.Palette 1,80:Palette.LayerB 1					// Palette for Layer B = 1
	Ink.Palette 2,48
	// Setup the Sprite Palette 3 - Is same as 2 but alters inks 68,69,73,78 for green/yellow cars
	For f=48 To 63
		Ink f+16,Ink.Red(f),Ink.Green(f),Ink.Blue(f)
	Next
	// Alter the Red & Blue for Green & Yellow for Palette 3
	Ink 66,21,31,21:Ink 68,10,31,10:Ink 69,0,27,1 :Ink 70,2,21,4
	Ink 72,31,31,21:Ink 74,31,30,12:Ink 77,29,27,0:Ink 78,19,18,0
	Ink.Palette 3,64
	BackColour 1
	Enable.Sprites
	// Last Thing to Set
	KeyMul.Set Key.Code.Space,Key.Code.CursorLeft,Key.Code.CursorRight,Key.Code.ESC,Key.Code.F1,0
	Set.Sprites 30,50								// Gameplay Sprites are 50 sprites starting at Sprite 30
	Enable.G9KInterrupt
	
	frameCount=1
EndF


Function GameLoop:Void()
	BackColour 1
	// Main Game Loop
	// Get Input & Move Player
	Update.Objects
	
	// Cars Changed Position Place
	If PosTrigger=1 Then
		PositionOrderChange
	EndIf
	// Game Time
	frameCount=frameCount+1
	If frameCount>49 Then
		frameCount=0
		TimeUpdate
	EndIf
	If KeyMul.Get(KeyMul.Scan5)Then
		Asm
		Ld 		hl,32
		Ld		c,l
		Ld		b,h
		Add		hl,hl		;*2
		Add		hl,bc		;*3
		Ld		bc,Quig_g9k_Ink_Data
		Add		hl,bc
		Di:halt
		Quigs
	EndIf
	// Quit Game
	If KeyMul.Get(KeyMul.Scan4) Then
		EverythingBlack
		Disable.Sprites
		Disable.G9KInterrupt
		pattCount=Load.Pattern.A(App.Path+"mmm001.qpt",0,48)
		pattCount= Meta.Load(App.Path+"map1.qmt")
		pattCount= Load.MapB(App.Path+"map1.qmb")
		pattCount= Load.MapA(App.Path+"map1.qma")
		LoadMenuSprites
		LoadMenuSprites_PressFire
		LoadMenuSprites_Select
		LoadMenuSprites_Cursor
		LoadMenuSprites_GetReady
		
		menustatus=6
		menuactive=1
		ShowMMLogo
		Set.Sprites 0,112
		Enable.G9KInterrupt
		Enable.Sprites
		Ink.Palette 2,48
		Ink.Palette 1,144:Palette.LayerA 1
		Ink.Palette 0,128:Palette.LayerB 0
	EndIf
EndF

Function SprayObject:Void()
	g=SprayStatus(Reference.Object)
	If OnScreen(SprayX(Reference.Object),SprayY(Reference.Object)) Then
		If g=6 Then
			Clear.MGroup Reference.Object+70,1
			Kill.Object Current.Object
		Else
			Display.MSprite 2,0							// Set  the Spray To Palette 2 and Visble
			Draw.MSprite Reference.Object+70,SpraySpr(Reference.Object)+Shr1(g)
			SprayStatus(Reference.Object)=g+1
		EndIf
	Else
		Clear.MGroup Reference.Object+70,1
		Kill.Object Current.Object
	EndIf
EndF

Function Car1_Control:Void()
	// Car 1 on ground driving mode
	If Car1_Doing=0 Then
		a=Car1_Rotated - Car1_DesRot
		c=Min(Abs(a),Car1_TurnMax)
		If c>0 Then
			If a And 8192 Then
				Car1_Turn = Min(Car1_Turn+Car1_TurnInc,c)
			Else
				Car1_Turn = Max(Car1_Turn-Car1_TurnInc,-c)
			EndIf
			Car1_DesSpeed=Car1_CornerSpeed
		Else
			Car1_Rotated=Car1_DesRot
			Car1_DesSpeed=Car1_TopSpeed
			Car1_Turn=0
		EndIf
		If SlowDown=1 Then
			// Slow down the car cos Player is stopped
			Car1_DesSpeed=500
		EndIf
		If Car1_Speed <Car1_DesSpeed Then
			Car1_Speed=Min(Car1_Speed+Car1_AccelSpeed,Car1_DesSpeed)
		Else
			Car1_Speed=Max(Car1_Speed+Car1_DecelSpeed,Car1_DesSpeed)
		EndIf
		Car1_Rotated=(Car1_Rotated +Car1_Turn) And 16383
	ElseIf Car1_Doing=12 Then
		;If Car1_Speed>32 Then
		;	Car1_Rotated=(Car1_Rotated +200) And 16383
		;EndIf
		Car1_Speed=Max(Car1_Speed-80,0)
	ElseIf Car1_Doing > 29 Then
		// Car1 is in air
		Car1_InAirVal=Min(Car1_InAirVal-Car1_InAirInc,0)
		If Car1_AirCount=0 Then
			Car1_InAirInc=Car1_InAirInc-1
			If Car1_InAirVal=0 Then
				If Car1_Doing>30 Then
					Car1_Doing=30
					Car1_InAirInc=1
					Car1_InAirVal=0
					Car1_AirCount=0
				Else
					Car1_Doing=0
					Clear.MGroup 67,1
				EndIf
			EndIf
		EndIf
		Car1_AirCount=(Car1_AirCount+1) And 1
	EndIf
	If Car1_Doing = 2 Then
		// Player backwards cos of Hit
		Car1_Speed = Max(Car1_Speed-350,0)				// Slow down cos hit object
		If Car1_Speed=0 Then
			Car1_Doing=0
			Car1_Rotated=(Car1_Rotated+8192) And 16383
		EndIf
	Else
		Car1_Facing=RotGroup(Shr8((Car1_Rotated + Shl4(Car1_Turn)) And 16383))
	EndIf
	// Speecis 0..8191 and Rotation is 0..16383
	c=Car1_X:d=Car1_Y
	If Car1_Speed>0 Then
		a=Shr8(Car1_Speed):b=Shr9(Car1_Rotated)
		Car1_X.Frac.Add AngleX(a,b)							//Add X for (Speed 0..31,Rotation 0..31)
		Car1_Y.Frac.Add AngleY(a,b)							//Add Y for (Speed 0..31,Rotation 0..31)
	EndIf
	
	//Okay what are we going over???
	Hit = Attr.MapA(Car1_X,Car1_Y)						//What is beneath Vehicle Pivot Point
	If Hit>0 Then
		// Less That 17 Direction marker
		If Hit<17 Then
			Car1_DesRot=Shl8(Shl3((Hit-1)And 7))
		ElseIf Hit>20 Then
			// Only check these if in Driving Mode
			If Car1_Doing=0 Then
				If Hit < 25 Then
				ElseIf Hit = 34 Then
					//Glue
					If Car1_Speed>1500 Then
						Car1_Speed=Car1_Speed-500
					EndIf
				ElseIf Hit=38 Then
					//Escalator Down
					If (Game.Counter And 3)=0 Then		//Every 4th frame
						Car1_Y=Car1_Y+1
					EndIf
				ElseIf Hit = 26 Then
					//Bump
					If Car1_Speed>1200 Then
						Car1_Doing=31 : Car1_InAirInc=1 : Car1_InAirVal=0 : Car1_AirCount=0
					EndIf
				ElseIf Hit = 27 Then
					//Small Hop
					If Car1_Speed>2000 Then
						Car1_Doing=31 : Car1_InAirInc=2 : Car1_InAirVal=0 : Car1_AirCount=0
					EndIf
				ElseIf Hit=28 Then
					// Large Jump
					If Car1_Speed>2000 Then
						Car1_Doing=33 : Car1_InAirInc=6 : Car1_InAirVal=0 : Car1_AirCount=0
					EndIf
				ElseIf Hit=35 Then
					// Oil Spill
					Car1_Rotated=(Car1_Rotated+Shr5(Car1_Speed)) And 16383
					Car1_TurnInc=Car1_TurnMax
				ElseIf Hit=30 Then
					// Driven off map
					Car1_Doing=11 : Car1_Count=0
				ElseIf Hit=33 Then
					// Hit Solid But Only in driving mode
					Car1_Doing=2
					Car1_X=c:Car1_Y=d
					Car1_Rotated=(Car1_Rotated+8192) And 16383
				EndIf
			EndIf
			// Hit Solid Object
			If Hit = 29 Then
				Car1_Doing=2
				Car1_X=c:Car1_Y=d
				Car1_Rotated=(Car1_Rotated+8192) And 16383
			ElseIf Hit>127 Then
				// Waypoint
				f=Hit-128
				If f=Car1_NextWayPoint Then
					Car1_NextWayPoint=Car1_NextWayPoint+1
					If Car1_NextWayPoint>=Level_WayPoints Then
						Car1_NextWayPoint=0
					EndIf
					Car1_WayPoint=f : Car1_SpawnX = Car1_X : Car1_SpawnY = Car1_Y : Car1_SpawnRotate = Car1_Rotated
					If f=0 Then
						Car1_Lap=Car1_Lap+1
						;If Car1_Lap>0 Then
						;	Car1_Doing=12
						;EndIf
					EndIf
					g=Pos(1)
					Pos(1)=Shl8(Shl5(Car1_Lap And 7)+f)+(255-Shr8(Game.Counter))
					SortRaceOrder(1)
					If g>Pos(0) Then
						If Pos(1)<Pos(0) Then
							PlayEffect(9,255)
						EndIf
					EndIf
					
				EndIf
			EndIf
		EndIf
	EndIf
	
	
	
	// Car 1 Draw Stuff
	Display.MSprite 2,0
	Car1_Collidable=0
	
	If Car1_Doing<11 Then
		// Normal Driving
		If OnScreen(Car1_X,Car1_Y) Then
			Draw.MGroup Car1_Facing+16,54
			Car1_Collidable=1
			// Leave Spray ? Cos Car Visible
			If (Game.Counter And 1)=1 Then
				If Abs(Car1_Turn)>64 Then
					a=FindEmpty.Object(20,10)
					If a>=0 Then
						f=a-20
						SprayStatus(f)=0
						SprayX(f)=Car1_X+OrigSprayX(Car1_Facing)
						SprayY(f)=Car1_Y+OrigSprayY(Car1_Facing)
						SpraySpr(f)=OrigSpraySpr(Car1_Facing)
						Spawn.Object (a,SprayObject.Addr,f)
					EndIf
				EndIf
			EndIf
		Else
			Clear.MGroup 54,4
		EndIf
	ElseIf Car1_Doing=12 Then
		// Finished race
		If OnScreen(Car1_X,Car1_Y) Then
			Car1_Collidable=1
			Draw.MGroup Car1_Facing+16,54
		Else
			Clear.MGroup 54,4
		EndIf
		Clear.MGroup 67,1
	ElseIf Car1_Doing>29 Then
		// Jumping
		b=Shr1(Abs(Car1_InAirVal))
		If OnScreen(Car1_X-b,Car1_Y-b) Then
			Draw.MGroup Car1_Facing+16,54
			If Hit<>30 Then
				If OnScreen(Car1_X-8,Car1_Y-8) Then
					Draw.MSprite 67,144+(Game.Counter And 1)
				EndIf
			Else
				Clear.MGroup 67,1
			EndIf
		Else
			Clear.MGroup 54,4
			Clear.MGroup 67,1
		EndIf
	ElseIf Car1_Doing=28 Then
		// Flashing
		Car1_Count=Max(Car1_Count-1,0)
		If Car1_Count=0 Then
			Car1_Doing=0
		EndIf
		If Car1_Count And 8 Then
			If OnScreen(Car1_X,Car1_Y) Then
				Draw.MGroup Car1_Facing+16,54
			Else
				Clear.MGroup 54,4
			EndIf
		Else
			Clear.MGroup 54,4
		EndIf
	ElseIf Car1_Doing=11 Then
		// Falling off map
		Car1_Speed = Max(Car1_Speed-500,0)				// Slow down cos off map
		Car1_Count=Car1_Count+1
		If Car1_Count>80 Then
			Car1_X=Car1_SpawnX:Car1_Y=Car1_SpawnY
			Car1_Rotated=0; Car1_SpawnRotate
			Car1_Speed = 0 : Car1_Doing=28 : Car1_Count=32
			Clear.MGroup 54,4
			If Car1_NextWayPoint=0 Then
				Car1_NextWayPoint=Level_WayPoints
				Car1_Lap=Car1_Lap-1
			Else
				Car1_NextWayPoint=Car1_NextWayPoint-1
			EndIf
		Else
			If OnScreen(Car1_X,Car1_Y) Then
				Display.MSprite 2,2					//Make Behind Background
				Clear.MGroup 54,4
				Draw.MGroup 76+(Shr3(Car1_Count)),54
			Else
				Clear.MGroup 54,4
			EndIf
		EndIf
	EndIf
EndF

Function SortRaceOrder:Void(a)
	n=Pos(a)
	b=12
	For f=0 To 3
		If f<>a Then
			If n<=Pos(f) Then
				b=b+15
			EndIf
		EndIf
	Next
	If b<>n Then
		n=PosNx(a)
		PosTrigger=1
		For f=0 To 3
			If f<>a Then
				If PosNx(f)=b Then
					If PosNx(f)<n Then
						PosNx(f)=PosNx(f)+15
					ElseIf PosNx(f)>n Then
						PosNx(f)=PosNx(f)-15
					EndIf
				Else
					If PosNx(f)>b Then
						If PosNx(f)<n Then
							PosNx(f)=PosNx(f)+15
						EndIf
					ElseIf PosNx(f)<b Then
						If PosNx(f)>n Then
							PosNx(f)=PosNx(f)-15
						EndIf
					EndIf
				EndIf
			EndIf
		Next
		PosNx(a)=b
	EndIf
EndF

					PlayEffect (14,255)					//Oil
					OilSkid=Game.Counter
				ElseIf OilSkid>Game.Counter Then
					OilSkid=0
				EndIf
			ElseIf Hit=30 Then
				// Driven off Map
				Player_Doing=11
				Player_Count=0
				SlowDown=1							//Slow other cars down cos we're off map
				PlayEffect (5,255)							//Falling
			ElseIf Hit=33 Then
				// Hit Solid But Only in driving mode
				Player_Doing=2
				Player_X=c:Player_Y=d
				Player_Rotated=(Player_Rotated+8192) And 16383
				If Player_Speed>3000 Then
					PlayEffect (12,255)					//Hit Solid
				ElseIf Player_Speed>1200 Then
					PlayEffect (13,64)
				EndIf
			EndIf
		EndIf
		// Hit Solid Object
		If Hit = 32 Then
			Player_Doing=2
			Player_Rotated=(Player_Rotated+8192) And 16383
			If Player_Speed>3000 Then
				PlayEffect (12,255)						//Hit Solid
			ElseIf Player_Speed>1400 Then
				PlayEffect (13,64)
			Else
				Player_Speed=1000
			EndIf
			Player_X=c:Player_Y=d
		EndIf
		// Waypoint
		If Hit>127 Then
			f=Hit-128
			If f=Player_NextWayPoint Then
				Player_NextWayPoint=Player_NextWayPoint+1
				If Player_NextWayPoint>=Level_WayPoints Then
					Player_NextWayPoint=0
				EndIf
				Player_WayPoint=f
				Player_SpawnX = Player_X
				Player_SpawnY =Player_Y
				Player_SpawnRotate = Player_Rotated
				If f=0 Then
					Player_Lap=Player_Lap+1
					Pattern.Sprite 31,245-Player_Lap
					;LapCountdownChange
					PlayEffect (10,255)
				EndIf
				g=PosNx(0)
				Pos(0)=Shl8(Shl5(Player_Lap And 7)+f)+(255-Shr8(Game.Counter))
				SortRaceOrder(0)
				If PosNx(0)<g Then
					PlayEffect(8,255)
				EndIf
				
			EndIf
		EndIf
	EndIf
	
	If Car1_Collidable Then
		If Collision(26,Player_X,Player_Y,Car1_X,Car1_Y) Then
			PlayerCollide(Car1_X,Car1_Y,Car1_Speed,Car1_Rotated)
			Car1_X=x1:Car1_Y=y1
			Car1_Speed=c1:Car1_Rotated=r1
		EndIf
	EndIf
	If Car2_Collidable Then
		If Collision(26,Player_X,Player_Y,Car2_X,Car2_Y) Then
			PlayerCollide(Car2_X,Car2_Y,Car2_Speed,Car2_Rotated)
			Car2_X=x1:Car2_Y=y1
			Car2_Speed=c1:Car2_Rotated=r1
		EndIf
	EndIf
	If Car3_Collidable Then
		If Collision(26,Player_X,Player_Y,Car3_X,Car3_Y) Then
			PlayerCollide(Car3_X,Car3_Y,Car3_Speed,Car3_Rotated)
			Car3_X=x1:Car3_Y=y1
			Car3_Speed=c1:Car3_Rotated=r1
		EndIf
	EndIf
	
	// Set Scrolls
	Scroll.LayerA Range(MapXMin,MapXMax,Player_X-128),Range(MapYMin,MapYMax,Player_Y-100)
	Scroll.LayerB Shr2(Scroll.LayerA.X),Shr2(Scroll.LayerA.Y)
	
	// Draw Player Car
	Display.MSprite 2,0
	If Player_Doing<11 Then
		//**** JUST DRIVING Doing<11 ****
		If OnScreen(Player_X,Player_Y) Then
			Draw.MGroup Player_Facing+32,50
			Clear.MGroup 66,1
		Else
			Clear.MGroup 50,4
		EndIf
	ElseIf Player_Doing>29 Then
		//**** IN AIR Doing>29 ****
		b=Shr1(Abs(Player_InAirVal))
		If OnScreen(Player_X-b,Player_Y-b) Then
			Draw.MGroup Player_Facing+32,50
			If Hit<>30 Then
				If OnScreen(Player_X-8,Player_Y-8) Then
					Draw.MSprite 66,144+(Game.Counter And 1)
				EndIf
			Else
				Clear.MGroup 66,1
			EndIf
		Else
			Clear.MGroup 50,4
			Clear.MGroup 66,1
		EndIf
	ElseIf Player_Doing=28 Then
		Player_Count=Max(Player_Count-1,0)
		If Player_Count=0 Then
			Player_Doing=0
			SlowDown=0
		EndIf
		If Player_Count And 8 Then
			If OnScreen(Player_X,Player_Y) Then
				Draw.MGroup Player_Facing+32,50
			Else
				
				Clear.MGroup 50,4
			EndIf
		Else
			If flasNoise<Game.Counter Then
				PlayEffect(2,255)				//Collide Noise Loud
				flasNoise=Game.Counter+40
			EndIf
			Clear.MGroup 50,4
		EndIf
	ElseIf Player_Doing=11 Then
		Player_Speed = Max(Player_Speed-500,0)				// Slow down cos off map
		Player_Count=Player_Count+1
		If Player_Count>80 Then
			// Okay We Die So Better Respawn
			Player_X=Player_SpawnX:Player_Y=Player_SpawnY
			Player_Rotated=Player_SpawnRotate
			Player_Speed = 0
			
			a=Range(MapXMin,MapXMax,Player_X-128)
			b=Range(MapYMin,MapYMax,Player_Y-100)
			
			Scroll.LayerA a-1,b-1
			Draw.LayerA
			a=Shr2(a):b=Shr2(b)
			Scroll.LayerB a-1,b-1
			Draw.LayerB
			
			Player_Turn=0
			Player_Doing=28
			Player_Count=64
			
			If Player_NextWayPoint=0 Then
				Player_NextWayPoint=Level_WayPoints
				Player_Lap=Player_Lap-1
			Else
				Player_NextWayPoint=Player_NextWayPoint-1
			EndIf
		EndIf
		Display.MSprite 2,2
		If OnScreen(Player_X,Player_Y) Then
			Clear.MGroup 50,4
			Draw.MGroup 64+(Shr3(Player_Count)),50
		Else
			Clear.MGroup 50,4
		EndIf
	EndIf
	
EndF

Function PlayerCollide:Void(x1,y1,c1,r1)
	
	If collideNoise<Game.Counter-50 Then
		If Player_Speed>1200 Then
			PlayEffect(16,255)			//Collide Noise Loud
		Else
			PlayEffect(18,180)			//Collide softer
		EndIf
		collideNoise=Game.Counter
	ElseIf collideNoise>Game.Counter Then
		collideNoise=0
	EndIf
	
	a=Shr1(Player_Speed+c1) : Player_Speed=a : c1=a
	If x1>Player_X Then
		a=1
	Else
		a=-1
	EndIf
	If y1>Player_Y Then
		b=1
	Else
		b=-1
	EndIf
	;x1=x1+a:y1=y1+b
	Player_X=Player_X-a:Player_Y=Player_Y-b
	While Collision(26,Player_X,Player_Y,x1,y1)
		x1=x1+a:y1=y1+b
		Player_X=Player_X-a:Player_Y=Player_Y-b
	Wend
	r1=r1+Shl9(a)
EndF

Function Load_Map:Void(Backname,ForeName,MapName,MapXMax,MapYMax)
	
	Var f:Int
	Var ts:Int
	Var fil:String
	
	// take info from String format "name=start,name=start"l
	While Backname>""
		f=FindStr(Backname,"=",1)
		fil=Mid(Backname,1,f-1)
		ts=CInt(Mid(Backname,f+1,999))
		f=Load.Pattern.B(App.Path+fil+".qpt",ts,80)
		f=FindStr(Backname,",",1)
		If f=0 Then
			Backname=""
		Else
			Backname=Mid(Backname,f+1,999)
		EndIf
	Wend
	// take info from String format "name=start,name=start"
	While ForeName>""
		f=FindStr(ForeName,"=",1)
		fil=Mid(ForeName,1,f-1)
		If f>0 Then
			ts=CInt(Mid(ForeName,f+1,999))
			f=Load.Pattern.A(App.Path+fil+".qpt",ts,32)
			f=FindStr(ForeName,",",1)
		EndIf
		If f>0 Then
			ForeName=Mid(ForeName,f+1,999)
		Else
			ForeName=""
		EndIf
	Wend
	
	// Load maps
	
	If Meta.Load(App.Path+MapName+".qmt")=0 Then
		;App.Exit
	EndIf
	If Load.MapB(App.Path+MapName+".qmb")=0 Then
		;App.Exit
	EndIf
	If Load.MapA(App.Path+MapName+".qma")=0 Then
		;App.Exit
	EndIf
	
	MapXMin = 0
	;	MapXMax = 2176+32
	MapXPMax = MapXMax+248
	MapYMin = 0
	;	MapYMax =2208
	MapYPMax = MapYMax+192
	
EndF
